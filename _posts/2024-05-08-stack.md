자료구조와 알고리즘 시간에 파이썬으로 작성한 예제
```python
a_count=list()

def push(push_date):
    a_count.append(push_date)

def pop():
    a_count.remove(a_count[-1])

a = input()
print(a)

for i in a:
    if i =='(':
        push(i)
    elif i ==')':
        pop()

if len(a_count)==0:
    print('OK')
else:
    print('error')
```

[while문으로 만들기]
```python
from collections import deque

BFS_graph = {
    'A': ['B', 'C'],
    'B': ['D', 'E'],
    'C': ['F', 'G'],
    'D': ['B'],  
    'E': ['B'],   
    'F': ['C'],  
    'G': ['C']   
}

visited_list = []

def bfs_func(BFS_graph, start):
    que = deque([start])

    while que:
        node = que.popleft()  
        print("node:", node)

        if node == "E":
            print("STOP")
            break

        if node not in visited_list:
            visited_list.append(node)
            print(f'방문지점: {visited_list}')
            que.extend(BFS_graph[node])  

bfs_func(BFS_graph, 'A')
```
